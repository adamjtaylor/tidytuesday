poll_differences %>% ggplot() +
  geom_segment(aes(x = `TRUE`, xend = `FALSE`, y = title, yend = title)) +
  geom_text(aes(x = ID, y = title), label = emoji("earth_africa"), colour = "grey") +
  geom_point(aes(x = `TRUE`, y = title), colour = "green") +
  geom_point(aes(x = `FALSE`, y = title), colour = "red") + theme_cowplot() +
  scale_x_reverse(breaks = c(1,25,50,100,126)) +
  scale_y_discrete(position = "right") +
  xlab("Rank") +
  theme(axis.title.y = element_blank(),
        axis.line.y = element_blank())
